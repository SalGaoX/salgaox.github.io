    自动检测并打印电子发票 基于Python 源代码

    萨尔丸子 2019/2/11 pythonPyPDF2win32

    # 程序会先检测剪切板中是否包含电子发票的网址，如有一致的则下载PDF分割后打印第一页PDF内容

    # 已下是代码块

    
# 版权所有  萨尔丸子 邮箱：34150740@qq.com
# 描述：
#     检测剪切板是否含有发票地址，如果包含则下载分割后打印第一页PDF
# 历史版本：
#     2019-2-11：创建 萨尔丸子

# -*- coding: UTF-8 -*-


import win32con
import win32clipboard as w
import time
import requests
import os
from PyPDF2 import PdfFileReader, PdfFileWriter
import win32print
import win32api


def getText():  # getText() 获取剪切板函数

    w.OpenClipboard()
    d = w.GetClipboardData(win32con.CF_UNICODETEXT)
    w.CloseClipboard()
    return d


def split_pdf(infile, out_path):  # split_pdf() 切割PDF函数
    """
    :param infile: 待拆分的pdf文件
    :param out_path: 拆分成单页的pdf文件的存储路径
    :return: 无
    """

    if not os.path.exists(out_path):
        os.makedirs(out_path)
    with open(infile, 'rb') as infile:

        reader = PdfFileReader(infile)
        number_of_pages = reader.getNumPages()  # 计算此PDF文件中的页数

        for i in range(number_of_pages):
            writer = PdfFileWriter()
            writer.addPage(reader.getPage(i))
            out_file_name = out_path + str(i + 1) + '.pdf'
            with open(out_file_name, 'wb') as outfile:
                writer.write(outfile)

print("创建列表L，用于存放已经打印过的地址")
L = []  # 创建列表L 用于存放已经打印过的地址
print("定义变量：dnum,fppdfmc,rfpdzjc")
dnum = 0  # 定义dnum 剪切板地址的长度
fppdfmc = 0  # 定义fppdfmc 下载的发票pdf文件名称
rfpdzjc = "einvoicelink.51fapiao.cn:8181"  # 定义rfpdzjc 正确地址检测
print("开始循环脚本")
for i in range(9999999999):  # 开始循环9999999999次
    time.sleep(1)  # 延迟1秒 命令：time.sleep(1) 以下不再重复备注
    a = getText()  # 获取剪切板赋值给变量a
    a = a[0:89]
    if a not in L:  # 判断变量a地址是否已经打印（已经存在与L列表），如果不存在则继续。末尾有else。以下代码都在IF判断下
        while dnum != 89:  # 长度不为89时暂停1秒重新判断，等于89长度时，判断为是发票地址则继续。
            time.sleep(1)
            a = getText()
            a = a[0:89]
            dnum = len(a)
        print("判断长度为 %s\n判断后符合发票地址长度" % (dnum))  # 长度符合，继续脚本
        time.sleep(1)
        fpdzjc = a.split("/", 5)[2]
        while fpdzjc != rfpdzjc:
            time.sleep(1)
            a = getText()
            a = a[0:89]
            fpdzjc = a.split("/", 5)[2]
        fpmc = a.split("/", 5)[5]
        fppdfmc = fpmc + ".pdf"
        print("发票PDF文件名为 %s" % (fppdfmc))
        print("下载开始下载PDF文件")
        r = requests.get(a)
        with open(fppdfmc, "wb") as code:
            code.write(r.content)
        time.sleep(1)
        in_File = fppdfmc
        out_Path = './'  # 生成输出文件夹
        print("开始分割PDF文件")
        split_pdf(in_File, out_Path)
        print("开始打印1.PDF")
        time.sleep(1)
        win32api.ShellExecute(0, 'print', '1.pdf', '', '', 0)
        print("将变量a写入已经打印列表L")
        L.append(a)
        print("已经打印过的发票列表为:\n%s" % (L))
    else:
        time.sleep(1)
