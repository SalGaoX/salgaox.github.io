    自动核对ERP-WMS-EBS 库存 基于Python 源代码

    萨尔丸子 2019/2/16 pythonnumpypandas

    # 自动核对目录下 （erp.xlsx， wms.xls， ebs.xls） 三个表格内的库存商品数量（多行相同代码会先求和） 并将结果写入erp-wms-ebs_库存核对结果.xlsx

    # 已下是代码块

    
#! /usr/bin/env python3
# -*- coding: utf-8 -*-


# 创建者：萨尔丸子             电子邮箱:34150740@qq.com
# 描述：
#   自动核对目录下 （erp.xlsx， wms.xls， ebs.xls） 三个表格内的库存商品数量 并将结果写入erp-wms-ebs_库存核对结果.xlsx
# 历史版本：
#   2019年2月16日：创建 萨尔丸子
#


import numpy as np
import pandas as pd
import os
import time

# 定义文件路径变量
erpFile = os.path.exists("erp.xlsx")
wmsFile = os.path.exists("wms.xls")
ebsFile = os.path.exists("ebs.xls")

# 判断所需三个文件是否存在

while erpFile == False or wmsFile == False or ebsFile == False:
    erpFile = os.path.exists("erp.xlsx")
    wmsFile = os.path.exists("wms.xls")
    ebsFile = os.path.exists("ebs.xls")
    if erpFile != True:
        print("所在目录未检测到erp.xlsx，请检查是否存在！5秒后继续检测")
        time.sleep(5)
    if wmsFile != True:
        print("所在目录未检测到wms.xls，请检查是否存在！5秒后继续检测")
        time.sleep(5)
    if ebsFile != True:
        print("所在目录未检测到ebs.xls，请检查是否存在！5秒后继续检测")
        time.sleep(5)

print("所需文件检测完毕，开始处理数据......")

# 读取三个系统库存表格
print("读取erp.xlsx文件数据......")
erp = pd.DataFrame(pd.read_excel("erp.xlsx", dtype={'商品数量': np.float}))
print("读取wms.xls文件数据......")
wms = pd.DataFrame(pd.read_excel("wms.xls", dtype={'库存数量': np.float}))
print("读取ebs.xls(从第三行开始读取)文件数据......")
ebs = pd.DataFrame(pd.read_excel("ebs.xls", dtype={'现有数量': np.float}, header=2))

# 更改标题名
print("更改erp.xlsx,商品数量: erp库存")
erp.rename(columns={"商品数量": "erp库存"}, inplace=True)
print("更改wms.xls,库存数量: wms库存")
wms.rename(columns={"库存数量": "wms库存"}, inplace=True)
print("更改ebs.xls,物料编码: 商品编码, 现有数量: ebs库存, 通用名: 商品名称")
ebs.rename(columns={"物料编码": "商品编码", "现有数量": "ebs库存", "通用名": "商品名称"}, inplace=True)

# 汇总数据
print("开始统计erp汇总库存......")
totalErp = pd.pivot_table(erp, index=['商品编码'], values=['erp库存'], aggfunc='sum')
print("开始统计wms汇总库存......")
totalWms = pd.pivot_table(wms, index=['商品编码'], values=['wms库存'], aggfunc='sum')
print("开始统计ebs汇总库存......")
totalEbs = pd.pivot_table(ebs, index=['商品编码'], values=['ebs库存'], aggfunc='sum')

# 合并数据erp为依据
print("开始以erp汇总库存为主表合并erp与wms汇总库存......")
erpWms = pd.merge(totalErp, totalWms, how='outer', on="商品编码")
print("开始以erp汇总库存为主表合并erp-wms-ebs汇总库存......")
finalStock = pd.merge(erpWms, totalEbs, how='outer', on="商品编码")

# 替换空值
finalStock = finalStock.fillna(0.0)

# 计算结果 erp为主
print("开始计算erp汇总库存减去wms汇总库存，并生成新的一列<erp减去wms库存>......")
finalStock["erp减去wms库存"] = finalStock["erp库存"] - finalStock['wms库存']
print("开始计算erp汇总库存减去ebs汇总库存，并生成新的一列<erp减去ebs库存>......")
finalStock["erp减去ebs库存"] = finalStock["erp库存"] - finalStock['ebs库存']
print("开始计算差异")
finalStockDifference = finalStock[~finalStock["erp减去wms库存"].isin([0]) | ~finalStock["erp减去ebs库存"].isin([0])]
print("计算完毕")
print("开始将结果写入到文件<erp-wms-ebs_库存核对结果.xlsx>中......")
print("所需时间根据表格大小有所不同，完成后会有提示，请勿退出软件！")

# 将结果写入文件
writer = pd.ExcelWriter("erp-wms-ebs_库存核对结果.xlsx")
finalStockDifference.to_excel(excel_writer=writer, sheet_name="核对有差异")
finalStock.to_excel(excel_writer=writer, sheet_name="核对结果")
writer.save()
writer.close()

# 倒计时10秒后退出
print("大功告成，请检查程序所在目录下<erp-wms-ebs_库存核对结果.xlsx>\n程序将在10秒后退出！")
t = 10
for i in range(10):
    print(t)
    time.sleep(1)
    t -= 1
