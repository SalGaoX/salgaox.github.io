import wx
import wx.grid
import xlrd
import xlwt
import os
import easygui as eg
import tkinter as tk
from tkinter import filedialog
import tkinter.messagebox
from PIL import Image
import openpyxl as op


excel_path = filedialog.askopenfilename(title='请选择基础表格', filetypes=[('xlsx', '*.xlsx')], initialdir='.\\')
if excel_path == '':
    tk.messagebox.showinfo('错误提示', '你未选择任何表格！')
pathname = os.path.basename(excel_path)
data = []
column_names = []
workbook = xlrd.open_workbook(excel_path)
sheetname = workbook.sheet_names()[0]
data_sheet = workbook.sheets()[0]
sheet1 = workbook.sheet_by_name(sheetname)
nrows = data_sheet.nrows
column_names = sheet1.row_values(0)
rows = sheet1.row_values(1)
for i in range(nrows):
    if i == 0:
        continue
    rows = sheet1.row_values(i)
    data.extend([rows])



class MyGridTable(wx.grid.GridTableBase):
    def __init__(self):
        super().__init__()
        self.colLabels = column_names

    def GetNumberRows(self):
        return len(data)

    def GetNumberCols(self):
        return len(data[0])

    def GetValue(self, row, col):
        return data[row][col]

    def GetColLabelValue(self, col):
        return self.colLabels[col]

    def SetValue(self, row, col, value):
        pass

            if Yes_or_No == '是':
                print("已收集，需要重新收集主图")
                root = tk.Tk()
                root.withdraw()
                file_path = filedialog.askopenfilename(filetypes=[('jpg', '*.jpg')])
                if file_path == '':
                    tk.messagebox.showinfo('错误提示', '你未选择任何图片！'.format(productCode))

                # 存在     True
                # 不存在   False

                    # 如果不存在则创建目录
                    # 创建目录操作函数

                else:

                    b = f.read()

                        copy_f.write(b)
                        print('图片拷贝成功')
                        image = Image.open('{0}\\1.jpg'.format(mkpath))
                        resized_image = image.resize((800, 800), Image.ANTIALIAS)
                        resized_image.save('{0}\\1.jpg'.format(mkpath))
                        wb = op.load_workbook(excel_path)
                        opSheet = wb['Sheet1']
                        for rowNum in range(2, opSheet.max_row + 1):
                            doneProductCode = opSheet.cell(row=rowNum, column=2).value
                            if doneProductCode == productCode:
                                opSheet.cell(row=rowNum, column=7).value = '1'

    def CreateGrid(self, parent):
        grid = wx.grid.Grid(parent)
        tablebase = MyGridTable()
        grid.SetTable(tablebase, True)
        # 设置行和列自定调整
        grid.AutoSize()

        return grid


class App(wx.App):
    def OnInit(self):
        # 创建窗口对象
        frame = MyFrame()
        frame.Show()
        return True
